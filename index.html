1. JavaScript Fundamentals

Variables and Data Types

// Variable declarations
let name = 'John';              // String
const age = 25;                 // Number
var isStudent = true;           // Boolean
let fruits = ['apple', 'banana']; // Array
let person = {                  // Object
    name: 'John',
    age: 25
};
let empty = null;              // Null
let undefined_var;             // Undefined
let symbol = Symbol('id');     // Symbol
let bigInt = 9007199254740991n; // BigInt

Type Conversion and Coercion

// Type conversion
String(123)          // "123"
Number("123")        // 123
Boolean(1)           // true

// Type coercion
"123" + 1           // "1231"
"123" - 1           // 122
true + 1            // 2

2. ES6+ Modern JavaScript

Let and Const

// Block scoping
{
    let blockScoped = 'only available here';
    const CONSTANT = 'cannot be reassigned';
}

// Template Literals
const name = 'John';
const greeting = `Hello ${name}!`;

// Arrow Functions
const add = (a, b) => a + b;
const multiply = (a, b) => {
    return a * b;
};

Destructuring

// Array destructuring
const [first, second, ...rest] = [1, 2, 3, 4, 5];

// Object destructuring
const { name, age, country = 'USA' } = person;

// Nested destructuring
const {
    address: { street, city }
} = complexObject;

Spread and Rest Operators

// Spread operator
const numbers = [1, 2, 3];
const newNumbers = [...numbers, 4, 5];

// Rest parameters
function sum(...numbers) {
    return numbers.reduce((acc, curr) => acc + curr, 0);
}

// Object spread
const baseConfig = { api: 'http://api.example.com' };
const config = { ...baseConfig, timeout: 5000 };

3. Advanced Functions

Function Types and Patterns

// Function declaration
function greet(name) {
    return `Hello ${name}`;
}

// Function expression
const greetExpr = function(name) {
    return `Hello ${name}`;
};

// Arrow function
const greetArrow = name => `Hello ${name}`;

// IIFE (Immediately Invoked Function Expression)
(function() {
    console.log('Executed immediately');
})();

Closures and Scope

function counter() {
    let count = 0;
    return {
        increment: () => ++count,
        decrement: () => --count,
        getCount: () => count
    };
}

const myCounter = counter();
myCounter.increment(); // 1
myCounter.increment(); // 2

Higher-Order Functions

// Function that returns a function
const multiply = (factor) => {
    return (number) => number * factor;
};

const double = multiply(2);
double(5); // 10

// Function that takes a function as argument
const applyOperation = (numbers, operation) => {
    return numbers.map(operation);
};

4. Advanced Arrays and Objects

Array Methods

const numbers = [1, 2, 3, 4, 5];

// map, filter, reduce
const doubled = numbers.map(n => n * 2);
const evens = numbers.filter(n => n % 2 === 0);
const sum = numbers.reduce((acc, curr) => acc + curr, 0);

// find, some, every
const found = numbers.find(n => n > 3);
const hasEven = numbers.some(n => n % 2 === 0);
const allPositive = numbers.every(n => n > 0);

// Array manipulation
const removed = numbers.splice(2, 1, 6);
const sliced = numbers.slice(1, 3);
const combined = [...numbers, ...evens];

// Object methods
Object.keys(person);

Object.entries(person);

// Object property descriptors
Object.defineProperty(person, 'id', {

5. Asynchronous JavaScript

Promises

// Creating a promise
const delay = ms => new Promise(resolve => setTimeout(resolve, ms));

// Using promises
delay(2000)
    .then(() => console.log('Delayed log'))
    .catch(error => console.error(error));

// Promise methods
Promise.all([promise1, promise2])
Promise.race([promise1, promise2])
Promise.allSettled([promise1, promise2])

Async/Await

async function fetchData() {
    try {
        const response = await fetch('https://api.example.com/data');
        const data = await response.json();
        return data;
    } catch (error) {
        console.error('Error:', error);
    }
}

// Using async/await with array methods
async function processUsers(userIds) {
    const promises = userIds.map(id => fetchUser(id));
    const users = await Promise.all(promises);
    return users;
}

6. Error Handling

// Try-catch
try {
    throw new Error('Something went wrong');
} catch (error) {
    console.error(error.message);
} finally {
    console.log('Cleanup code');
}

// Custom errors
class ValidationError extends Error {
    constructor(message) {
        super(message);
        this.name = 'ValidationError';
    }
}

// Async error handling
async function fetchData() {
    try {
        const response = await fetch('https://api.example.com/data');
        if (!response.ok) {
            throw new Error(`HTTP error! status: ${response.status}`);
        }
        const data = await response.json();
        return data;
    } catch (error) {
        console.error('Error:', error);
        throw error; // Re-throwing the error
    }
}
